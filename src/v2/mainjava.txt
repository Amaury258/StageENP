public static void main(String[] args) {
        int etat = 1;
        boolean prompt = false;
        while(etat>S_ETAT_ERREUR){
            switch(etat){
                case S_ETAT_INIT:
                    if(!prompt){
                        System.out.println("Recherche de périphérique USB");
                        prompt = true;
                    }
                    if(etat == S_ETAT_INIT){
                        //System.out.println("Aucun périphérique USB détecté");
                        try {
                            Thread.sleep(1000);
                        } catch (InterruptedException e) {
                            // TODO Auto-generated catch block
                            e.printStackTrace();
                        }
                    }
                    break;
                case S_ETAT_DETECTE:
                    System.out.println("Clé détecté");
                    //Verification avec la base des périphériques autorisés
                    etat = S_ETAT_AUTORISE;
                    break;
                case S_ETAT_AUTORISE:
                    System.out.println("Autorisé");
                    //Analyse antivirus
                    etat = S_ETAT_AUCUN_VIRUS;
                    break;
                case S_ETAT_AUCUN_VIRUS:
                    System.out.println("Aucun antivirus détecté");
                    etat = S_ETAT_INIT;
                    break;
                case S_ETAT_EJECTE:
                    System.out.println("Éjection");
                    etat = S_ETAT_ERREUR;
                    break;
            }
        }
